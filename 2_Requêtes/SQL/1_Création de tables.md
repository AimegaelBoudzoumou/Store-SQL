```sql
CREATE TABLE employes (
    idValue NUMBER GENERATED BY DEFAULT AS IDENTITY,
    matricule VARCHAR2(50),
    nom VARCHAR2(50),
    prenom VARCHAR2(50),
    full_name VARCHAR2(150) GENERATED ALWAYS AS (
    	prenom || ' ' || nom    
    ),
    PRIMARY KEY(matricule)
);
```
